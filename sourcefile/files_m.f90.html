      ! Auxiliary functions


      FUNCTION c_to_f(c_string) RESULT(res)
         ! Auxiliary function used as interface to convert c strings to fortran strings
         IMPLICIT NONE
         CHARACTER(C_CHAR)              :: c_string
         CHARACTER(LEN=:), ALLOCATABLE  :: res
         INTEGER :: l !length
         l = 1
         DO WHILE (c_string(l:l) /= C_NULL_CHAR)
            l = l + 1
         END DO
         res = c_string(1:l-1)
      END FUNCTION c_to_f
